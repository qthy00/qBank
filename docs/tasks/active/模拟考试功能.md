## 技术方案

### 页面路由

```
/exam                 # 模拟考试首页（选择题库）
/exam/[id]            # 具体模拟考试（使用ExamMock组件）
/exam/report/[id]     # 考试报告
```

### API规划

```typescript
// api/exam/index.ts

// 生成模拟试卷
export const generateMockPaper = (data: MockPaperReqVO) => {
  return httpPost('GenerateMockPaper', '/member/exam/paper/generate', data)
}

// 提交试卷
export const submitMockPaper = (data: SubmitPaperVO) => {
  return httpPost('SubmitMockPaper', '/member/exam/paper/submit', data)
}

// 获取考试报告
export const getExamReport = (paperId: number) => {
  return httpGet('ExamReport', `/member/exam/paper/report/${paperId}`)
}

// 获取考试历史
export const getExamHistory = (params: any) => {
  return httpGet('ExamHistory', '/member/exam/paper/history', { query: params })
}
```

### 类型定义

```typescript
// types/exam/index.d.ts

export interface MockPaperReqVO {
  qbankId: number
  subjectId?: number
  durationMinutes: number
  questionCount: number
  questionTypes: number[]
  difficulty?: number
  randomQuestions?: boolean
}

export interface SubmitPaperVO {
  paperId: number
  answers: {
    questionId: number
    answer: string
  }[]
  spendTime: number
}

export interface ExamReportVO {
  paperId: number
  score: number
  totalScore: number
  accuracy: number
  spendTime: number
  rank?: string
  questionAnalysis: QuestionAnalysis[]
}
```
