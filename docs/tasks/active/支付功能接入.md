## 任务描述

实现订单支付功能，支持微信支付（服务号支付/扫码支付）、支付回调处理。

---

## 实现方案

### 最终方案：微信服务号支付 + 微信扫码支付

- **服务号支付**: 在微信内置浏览器中使用服务号支付
- **扫码支付**: 在PC浏览器中生成二维码扫码支付
- **解决方式**: 不采用H5支付，直接使用服务号支付能力

// api/pay/index.ts - 已实现

// 创建支付订单
export const createPayment = (data: PaymentCreateReqVO) => {
  return httpPost('CreatePayment', '/member/pay/create', data)
}

// 查询支付状态
export const getPaymentStatus = (orderId: number) => {
  return httpGet('PaymentStatus', '/member/pay/status', { query: { orderId } })
}

// 处理支付回调
export const handlePayNotify = (data: PayNotifyVO) => {
  return httpPost('PayNotify', '/member/pay/notify', data)
}
